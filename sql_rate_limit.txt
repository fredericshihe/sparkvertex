
-- Create table for tracking API usage
create table if not exists user_api_limits (
  user_id uuid references auth.users not null,
  endpoint text not null,
  minute_count int default 0,
  last_request_at timestamptz default now(),
  daily_count int default 0,
  last_date date default current_date,
  primary key (user_id, endpoint)
);

-- Enable RLS
alter table user_api_limits enable row level security;

-- Policies
create policy "Users can view their own limits"
  on user_api_limits for select
  using (auth.uid() = user_id);

create policy "Users can update their own limits"
  on user_api_limits for update
  using (auth.uid() = user_id);

create policy "Users can insert their own limits"
  on user_api_limits for insert
  with check (auth.uid() = user_id);
